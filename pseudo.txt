// Начало программы
// Установить локаль для поддержки русского языка
Установить локаль "Russian"

// Определить размер матрицы
N = 5

// Инициализировать матрицы
Создать матрицу matrix[N][N]
Создать матрицу resultMatrix[N][N]

// Инициализировать переменные для значений диагоналей
mainDiagonalValue = 0
aboveDiagonalValue = 0
belowDiagonalValue = 0

// Определить меню программы
menu = [
    "====================================",
    "=== ВОЗМОЖНЫЕ ДЕЙСТВИЯ ПРОГРАММЫ ===",
    "====================================",
    "1. Ввод элементов матрицы",
    "2. Вывод матрицы на экран",
    "3. Сортировка элементов строк матрицы по возрастанию",
    "4. Сортировка элементов строк матрицы по убыванию",
    "5. Запись элементов матрицы в файл",
    "6. Чтение чисел из файла в матрицу",
    "7. Сумма диагоналей",
    "8. Выход из программы"
]

// Основной цикл программы
Цикл (бесконечный)
    // Вывести меню на экран
    Для каждой строки в menu
        Вывести строку

    // Запросить выбор действия у пользователя
    Вывести "Выберите действие: "
    Ввести choice

    // Обработка выбора пользователя
    Если choice == 1 тогда
        // Ввод элементов матрицы
        Вывести "Введите элементы матрицы 5x5 (по строкам):"
        Для i от 0 до N-1
            Для j от 0 до N-1
                Ввести matrix[i][j]
    Иначе если choice == 2 тогда
        // Вывод матрицы на экран
        Вывести "\nТекущая матрица:"
        Вызвать функцию printMatrix(matrix)
    Иначе если choice == 3 тогда
        // Сортировка элементов строк матрицы по возрастанию
        Для i от 0 до N-1
            Вызвать функцию sortRowAscending(matrix[i])
        Вывести "Строки матрицы отсортированы по возрастанию."
    Иначе если choice == 4 тогда
        // Сортировка элементов строк матрицы по убыванию
        Для i от 0 до N-1
            Вызвать функцию sortRowDescending(matrix[i])
        Вывести "Строки матрицы отсортированы по убыванию."
    Иначе если choice == 5 тогда
        // Запись элементов матрицы в файл
        Открыть файл "matrix.txt" для записи
        Если файл открыт успешно тогда
            Для i от 0 до N-1
                Для j от 0 до N-1
                    Записать matrix[i][j] в файл
            Закрыть файл
            Вывести "Матрица записана в файл 'matrix.txt'."
        Иначе
            Вывести "Ошибка открытия файла для записи!"
    Иначе если choice == 6 тогда
        // Чтение чисел из файла в матрицу
        Открыть файл "matrix.txt" для чтения
        Если файл открыт успешно тогда
            Для i от 0 до N-1
                Для j от 0 до N-1
                    Читать matrix[i][j] из файла
            Закрыть файл
            Вывести "Матрица считана из файла 'matrix.txt'."
        Иначе
            Вывести "Ошибка открытия файла для чтения!"
    Иначе если choice == 7 тогда
        // Сумма диагоналей
        Если matrix пустая тогда
            Вывести "Введите элементы матрицы 5x5 (по строкам):"
            Для i от 0 до N-1
                Для j от 0 до N-1
                    Ввести matrix[i][j]

        Вывести "Введите значение для добавления к элементам главной диагонали: "
        Ввести mainDiagonalValue
        Вывести "Введите значение для добавления к элементам выше главной диагонали: "
        Ввести aboveDiagonalValue
        Вывести "Введите значение для добавления к элементам ниже главной диагонали: "
        Ввести belowDiagonalValue

        Для i от 0 до N-1
            Для j от 0 до N-1
                resultMatrix[i][j] = matrix[i][j]
                Если i == j тогда
                    resultMatrix[i][j] += mainDiagonalValue
                Если i < j тогда
                    resultMatrix[i][j] += aboveDiagonalValue
                Если i > j тогда
                    resultMatrix[i][j] += belowDiagonalValue

        Вывести "\nИсходная матрица:"
        Вызвать функцию printMatrix(matrix)
        Вывести "\nРезультирующая матрица:"
        Вызвать функцию printMatrix(resultMatrix)

        // Запись сумм в файл
        Открыть файл "sums.txt" для записи
        Если файл открыт успешно тогда
            sum = 0
            sumResult = 0
            Для i от 0 до N-1
                Для j от 0 до N-1
                    sumResult += resultMatrix[i][j]
                    sum += matrix[i][j]
            Записать "Сумма всех элементов результирующей матрицы: " + sumResult в файл
            Записать "Сумма всех элементов исходной матрицы: " + sum в файл
            Закрыть файл
            Вывести "Суммы записаны в файл 'sums.txt'."
        Иначе
            Вывести "Ошибка открытия файла для записи!"

        // Чтение из файла и вывод содержимого
        Открыть файл "sums.txt" для чтения
        Если файл открыт успешно тогда
            Пока есть строка в файле
                Читать line из файла
                Вывести line
            Закрыть файл
        Иначе
            Вывести "Ошибка открытия файла для чтения!"
    Иначе если choice == 8 тогда
        // Выход из программы
        Вывести "До свидания!"
        Выйти из цикла
    Иначе
        // Неверный выбор
        Вывести "Неверный выбор. Пожалуйста, попробуйте снова."
    Конец Если

    Вывести пустую строку
Конец Цикла
// Конец программы
